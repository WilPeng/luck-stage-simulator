## 一、基础规范

### 1.1 技术栈

- **框架**: Vue 3 + TypeScript
- **状态管理**: Pinia
- **UI 组件库**: TDesign Vue Next
- **路由**: Vue Router
- **HTTP 客户端**: Axios（封装）

### 1.2 项目结构

```
src/
├── api/                  # API 封装（已废弃，统一使用 services/api.ts）
├── assets/              # 静态资源
├── components/          # 公共组件
├── layouts/             # 布局组件
├── mock/                # Mock 数据
├── router/              # 路由配置
├── services/            # API 服务层
│   ├── api.ts          # 所有 API 调用（重要！）
│   └── mockApi.ts      # Mock API
├── stores/              # Pinia Store
├── types/               # 类型定义
│   ├── index.ts        # 统一导出
│   ├── season.ts       # 赛季相关类型
│   ├── round.ts        # 轮次相关类型
│   └── ...
```

```vue
<!-- views/player/PlayerTeamView.vue -->

<template>
  <!-- 1. 状态提示（future 时显示） -->
  <div v-if="pageStatus === 'future'" class="future-notice">
    <p>该阶段尚未开放</p>
  </div>

  <div v-else-if="pageStatus === 'completed'" class="completed-view">

  </div>

    <!-- 可操作内容 -->

import { computed } from 'vue'
import { useSeasonStore } from '@/stores/seasonStore'
import type { StageType } from '@/types/season'

// 从 URL 获取参数
const route = useRoute()
const round = computed(() => Number(route.params.round))
const stage = computed(() => route.params.stage as StageType)

// 获取页面状态（⚠️ 必须使用 Store 方法，禁止自行计算）
const seasonStore = useSeasonStore()
const pageStatus = computed(() => seasonStore.getStageStatus(round.value, stage.value))
</script>
```

### 6.2 菜单状态显示

```vue
<!-- PlayerLayout.vue -->

<template>
  <div
    v-for="stage in stages"
    :key="stage.type"
    class="nav-item"
    :class="{
      'status-completed': isCompleted(stage.type),
      'status-current': isCurrent(stage.type),
      'status-future': isFuture(stage.type)
    }"
  >
    {{ stage.name }}
    <span v-if="isCompleted(stage.type)">✓</span>
    <span v-if="isCurrent(stage.type)" class="pulse">●</span>
  </div>
</template>

<script setup lang="ts">
import { useSeasonStore } from '@/stores/seasonStore'

const seasonStore = useSeasonStore()

function isCompleted(stage: StageType) {
  return seasonStore.getStageStatus(round.value, stage) === 'completed'
}

function isCurrent(stage: StageType) {
  return seasonStore.getStageStatus(round.value, stage) === 'current'
}

function isFuture(stage: StageType) {
  return seasonStore.getStageStatus(round.value, stage) === 'future'
}
</script>
```

### 6.3 菜单点击规则

```typescript
// future 状态时点击无效
function handleNavClick(event: Event, stage: StageType) {
  if (seasonStore.getStageStatus(round.value, stage) === 'future') {
    event.preventDefault()
    MessagePlugin.warning('该阶段尚未开放')
    return
  }
  // 允许跳转
  router.push(getStagePath(round.value, stage))
}
```

### 8.2 矩阵状态样式

```css
.cell-completed { background: #f0fbf6; color: #00a870; }  /* 绿色 */
```

### 8.3 点击逻辑

```typescript
function handleCellClick(round: number, stage: StageType) {
  const status = seasonStore.getStageStatus(round, stage)

  if (status === 'current') {
    MessagePlugin.info('这是当前阶段')
    return
  }

  if (status === 'completed') {
    MessagePlugin.warning('已完成阶段不可切换')
    return
  }

  // future 状态时才允许切换
  // 只能切换到相邻的阶段
  // 发送 POST /api/admin/progress/set
  await seasonStore.setStage(round, stage)
}
```
